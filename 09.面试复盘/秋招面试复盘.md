    ```js
    fs.readFile('/etc/passwd', (err, data) => {
      if (err) throw err;
      console.log(data);
    });

    const readFile = promisify(fs.readFile)

    readFile('/etc/passwd').then((data) => {}, (err) => {})

    // 面试官提示
    promisify(fn)

    fn = (...args, (err, data) => {
    })

    // 实现promisify

    function promisify(fn) {
        return function () {
            let args = [...arguments];
            return new Promise((resolve, reject) => {
                fn(...args, (err, data) => {
                  if (err) reject(err);
                  resolve(data);
                })
            })
        }
    }
    ```
